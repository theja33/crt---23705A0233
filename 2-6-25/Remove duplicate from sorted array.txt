public class App{
    public static void main(String[] args) {
        int[]nums={1,2,2,3,3,5,6,6,6,7};
        int i=0;
        for(int j=0;j<nums.length;j++){
            if(nums[i]!=nums[j]){
                nums[i+1]=nums[j];
                i++;
                
            }
        }
    for(int j=0;j<=i;j++){
        System.out.print(nums[j]+" ");
    }
       
    }
}
